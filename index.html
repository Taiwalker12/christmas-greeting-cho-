import pygame import sys # Khởi tạo pygame pygame.init() # Kích thước của cửa sổ width, height = 600, 400 screen = pygame.display.set_mode((width, height)) pygame.display.set_caption("Chúc Mừng Giáng Sinh!") # Màu sắc WHITE = (255, 255, 255) BLACK = (0, 0, 0) RED = (255, 0, 0) # Font chữ font = pygame.font.Font(None, 74) # Các vị trí của văn bản text_yeu = font.render("Yêu chồng Tài không?", True, BLACK) text_co = font.render("Có", True, RED) text_khong = font.render("Không", True, RED) # Vị trí để vẽ các chữ yeu_rect = text_yeu.get_rect(center=(width // 2, height // 3)) co_rect = text_co.get_rect(center=(width // 2 - 100, height // 2)) khong_rect = text_khong.get_rect(center=(width // 2 + 100, height // 2)) # Hàm vẽ hình trái tim def draw_heart(): pygame.draw.polygon(screen, RED, [(width // 2, height // 3), (width // 2 + 40, height // 3 - 30), (width // 2 + 80, height // 3), (width // 2, height // 3 + 50), (width // 2 - 80, height // 3), (width // 2 - 40, height // 3 - 30)], 0) pygame.draw.circle(screen, RED, (width // 2 - 40, height // 3 - 30), 40) pygame.draw.circle(screen, RED, (width // 2 + 40, height // 3 - 30), 40) # Hàm chính để chạy chương trình def run_game(): running = True selected = None # Biến lưu trạng thái lựa chọn while running: screen.fill(WHITE) # Kiểm tra sự kiện di chuyển chuột mouse_x, mouse_y = pygame.mouse.get_pos() # Nếu chuột vào khu vực của "Không", đổi thành "Có" if khong_rect.collidepoint(mouse_x, mouse_y): text_khong = font.render("Có", True, RED) else: text_khong = font.render("Không", True, RED) # Vẽ các chữ lên màn hình screen.blit(text_yeu, yeu_rect) screen.blit(text_co, co_rect) screen.blit(text_khong, khong_rect) # Kiểm tra sự kiện chọn "Có" hay "Không" for event in pygame.event.get(): if event.type == pygame.QUIT: running = False if event.type == pygame.MOUSEBUTTONDOWN: if co_rect.collidepoint(mouse_x, mouse_y): selected = "Có" screen.fill(RED) # Nền đỏ khi chọn "Có" draw_heart() # Vẽ trái tim pygame.display.flip() # Cập nhật màn hình time.sleep(2) # Hiển thị trái tim trong 2 giây elif khong_rect.collidepoint(mouse_x, mouse_y): selected = "Không" screen.fill(WHITE) # Quay lại màn hình trắng pygame.display.flip() # Cập nhật màn hình # Nếu đã chọn "Có", vẽ trái tim if selected == "Có": draw_heart() pygame.display.flip() pygame.quit() sys.exit() # Chạy chương trình if __name__ == "__main__": run_game()
